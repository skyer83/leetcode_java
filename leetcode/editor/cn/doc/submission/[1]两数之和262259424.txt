class Solution {
    public int[] twoSum(int[] nums, int target) {
        if (nums == null || nums.length < 2) return null;

        int lastLen = nums.length;
        int firstLen = lastLen - 1;
        for (int i = 0; i < firstLen; i++) {
            for (int j = i + 1; j < lastLen; j++) {
                if (nums[i] + nums[j] == target) {
                    return new int[]{i, j};
                }
            }
        }
        return null;
    }
}
