class Solution {
    public int[] twoSum(int[] nums, int target) {
        int len = nums.length;
        Map<Integer, Integer> diffMap = new HashMap<>();
        for (int i = 0; i < len; i++) {
            int num = nums[i];
            Integer indexPosition = diffMap.get(num);
            if (indexPosition != null) {
                return new int[]{indexPosition, i};
            }
            diffMap.put(target - num, i);
        }
        return null;
    }
}
